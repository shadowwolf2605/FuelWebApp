import SwiftUI
import CoreLocation

// MARK: - Location Manager

final class LocationManager: NSObject, ObservableObject, CLLocationManagerDelegate {
    private let clManager = CLLocationManager()

    @Published var isTracking = false
    @Published var distanceMeters: Double = 0
    @Published var authorizationStatus: CLAuthorizationStatus = .notDetermined
    @Published var errorMessage: String?

    private var lastLocation: CLLocation?

    override init() {
        super.init()
        clManager.delegate = self
        clManager.desiredAccuracy = kCLLocationAccuracyBestForNavigation
        clManager.distanceFilter = 5 // update every 5 metres
        authorizationStatus = clManager.authorizationStatus
    }

    var distanceKm: Double { distanceMeters / 1000 }

    var canTrack: Bool {
        authorizationStatus == .authorizedWhenInUse || authorizationStatus == .authorizedAlways
    }

    func requestPermission() {
        clManager.requestWhenInUseAuthorization()
    }

    func startTrip() {
        distanceMeters = 0
        lastLocation = nil
        errorMessage = nil
        isTracking = true
        clManager.startUpdatingLocation()
    }

    /// Stops tracking and returns total km.
    @discardableResult
    func stopTrip() -> Double {
        isTracking = false
        clManager.stopUpdatingLocation()
        lastLocation = nil
        return distanceKm
    }

    // MARK: CLLocationManagerDelegate

    func locationManager(_ manager: CLLocationManager, didUpdateLocations locations: [CLLocation]) {
        guard let loc = locations.last,
              loc.horizontalAccuracy >= 0,
              loc.horizontalAccuracy <= 65 else { return }

        if let last = lastLocation {
            let delta = loc.distance(from: last)
            if delta > 0 {
                DispatchQueue.main.async {
                    withAnimation(.linear(duration: 0.25)) { self.distanceMeters += delta }
                }
            }
        }
        lastLocation = loc
    }

    func locationManagerDidChangeAuthorization(_ manager: CLLocationManager) {
        DispatchQueue.main.async { self.authorizationStatus = manager.authorizationStatus }
    }

    func locationManager(_ manager: CLLocationManager, didFailWithError error: Error) {
        DispatchQueue.main.async { self.errorMessage = error.localizedDescription }
    }
}

// MARK: - Model

struct FuelRecord: Identifiable {
    let id = UUID()
    let date: Date
    let startKm: Double
    let endKm: Double
    let liters: Double
    let pricePerLiter: Double

    var distance: Double   { endKm - startKm }
    var consumption: Double { (liters / distance) * 100 }
    var totalCost: Double  { liters * pricePerLiter }

    var dateFormatted: String {
        let f = DateFormatter()
        f.dateStyle = .medium
        f.timeStyle = .short
        f.locale = Locale(identifier: "bg_BG")
        return f.string(from: date)
    }
}

// MARK: - ViewModel

class FuelViewModel: ObservableObject {
    @Published var startKmText = ""
    @Published var endKmText   = ""
    @Published var litersText  = ""
    @Published var priceText   = ""
    @Published var history: [FuelRecord] = []

    private func parse(_ s: String) -> Double? {
        Double(s.replacingOccurrences(of: ",", with: "."))
    }

    var startKm: Double? { parse(startKmText) }
    var endKm: Double?   { parse(endKmText) }
    var liters: Double?  { parse(litersText) }
    var price: Double?   { parse(priceText) }

    var distance: Double? {
        guard let s = startKm, let e = endKm, e > s else { return nil }
        return e - s
    }

    var distanceError: Bool {
        guard let s = startKm, let e = endKm else { return false }
        return e <= s
    }

    var consumption: Double? {
        guard let d = distance, let l = liters, d > 0, l > 0 else { return nil }
        return (l / d) * 100
    }

    var totalCost: Double? {
        guard let l = liters, let p = price, l > 0, p > 0 else { return nil }
        return l * p
    }

    var isValid: Bool { consumption != nil && totalCost != nil }

    func save() {
        guard let s = startKm, let e = endKm, let l = liters, let p = price, isValid else { return }
        let record = FuelRecord(date: Date(), startKm: s, endKm: e, liters: l, pricePerLiter: p)
        withAnimation(.spring(response: 0.4, dampingFraction: 0.7)) {
            history.insert(record, at: 0)
        }
        startKmText = ""; endKmText = ""; litersText = ""; priceText = ""
    }

    func delete(_ record: FuelRecord) {
        withAnimation { history.removeAll { $0.id == record.id } }
    }
}

// MARK: - Root View

struct ContentView: View {
    @StateObject private var vm  = FuelViewModel()
    @StateObject private var gps = LocationManager()
    @Environment(\.colorScheme) var colorScheme

    var body: some View {
        NavigationStack {
            ScrollView {
                VStack(spacing: 20) {
                    GPSTripCard(vm: vm, gps: gps)
                    InputCard(vm: vm)
                    ResultCard(vm: vm)
                    HistorySection(vm: vm)
                }
                .padding(.horizontal, 16)
                .padding(.bottom, 40)
            }
            .background(Color(.systemGroupedBackground))
            .navigationTitle("Разход на гориво")
            .navigationBarTitleDisplayMode(.large)
            .toolbar {
                ToolbarItem(placement: .topBarTrailing) {
                    Image(systemName: colorScheme == .dark ? "moon.fill" : "sun.max.fill")
                        .foregroundStyle(colorScheme == .dark ? .indigo : .orange)
                }
            }
        }
    }
}

// MARK: - GPS Trip Card

struct GPSTripCard: View {
    @ObservedObject var vm:  FuelViewModel
    @ObservedObject var gps: LocationManager

    var body: some View {
        CardContainer {
            VStack(alignment: .leading, spacing: 16) {

                // Header
                HStack {
                    Label("GPS Проследяване", systemImage: "location.fill")
                        .font(.headline)
                    Spacer()
                    if gps.isTracking {
                        Text("АКТИВНО")
                            .font(.caption2.weight(.bold))
                            .foregroundStyle(.white)
                            .padding(.horizontal, 8).padding(.vertical, 3)
                            .background(Color.green).clipShape(Capsule())
                    }
                }

                // Live distance
                HStack(alignment: .center, spacing: 20) {
                    VStack(alignment: .leading, spacing: 4) {
                        Text("Изминато разстояние")
                            .font(.caption).foregroundStyle(.secondary)
                        HStack(alignment: .lastTextBaseline, spacing: 4) {
                            Text(String(format: "%.3f", gps.distanceKm))
                                .font(.system(size: 38, weight: .bold, design: .rounded))
                                .foregroundStyle(gps.isTracking ? .blue : .primary)
                                .contentTransition(.numericText(countsDown: false))
                                .animation(.linear(duration: 0.25), value: gps.distanceKm)
                            Text("км")
                                .font(.title2).foregroundStyle(.secondary)
                        }
                    }
                    Spacer()
                    if gps.isTracking { PulsingDot() }
                    else {
                        Image(systemName: "location.circle")
                            .font(.system(size: 48))
                            .foregroundStyle(Color(.tertiaryLabel))
                    }
                }

                // Error
                if let err = gps.errorMessage {
                    Label(err, systemImage: "exclamationmark.triangle.fill")
                        .font(.caption).foregroundStyle(.orange)
                        .padding(8)
                        .background(Color.orange.opacity(0.1))
                        .clipShape(RoundedRectangle(cornerRadius: 8))
                }

                // Buttons
                Group {
                    if !gps.canTrack {
                        TripButton(label: "Разреши достъп до GPS",
                                   icon: "location.fill", color: .blue,
                                   action: gps.requestPermission)
                    } else if gps.isTracking {
                        TripButton(label: "Спри пътуването",
                                   icon: "stop.fill", color: .red,
                                   action: stopTrip)
                    } else {
                        TripButton(label: "Стартирай пътуването",
                                   icon: "play.fill", color: .green,
                                   action: gps.startTrip)
                    }
                }
                .animation(.easeInOut(duration: 0.2), value: gps.isTracking)
                .animation(.easeInOut(duration: 0.2), value: gps.canTrack)
            }
        }
    }

    private func stopTrip() {
        let km = gps.stopTrip()
        guard km > 0.01 else { return }
        let rounded = (km * 10).rounded() / 10
        withAnimation {
            if let start = Double(vm.startKmText.replacingOccurrences(of: ",", with: ".")) {
                vm.endKmText = String(format: "%.1f", start + rounded)
            } else {
                vm.endKmText = String(format: "%.1f", rounded)
            }
        }
    }
}

            Circle().fill(Color.blue.opacity(0.25))
                .frame(width: 56, height: 56)
                .scaleEffect(animate ? 1.6 : 1.0)

                .frame(width: 36, height: 36)
                .scaleEffect(animate ? 1.25 : 1.0)
                .opacity(animate ? 0.2 : 0.8)

                .font(.title3).foregroundStyle(.white)
                .frame(width: 24, height: 24)

                animate = true
            }
        }
    }
}

// MARK: - Input Card

struct InputCard: View {
    @ObservedObject var vm: FuelViewModel
    @FocusState private var focused: InputField?
    enum InputField: Hashable { case startKm, endKm, liters, price }

    var body: some View {
        CardContainer {
            VStack(alignment: .leading, spacing: 18) {
                Label("Въведи данни", systemImage: "fuelpump.fill")
                    .font(.headline)

                // Start KM
                InputRow(icon: "flag.fill", iconColor: .green,
                         label: "Начален километраж", placeholder: "напр. 45000", unit: "км",
                         text: $vm.startKmText)
                    .focused($focused, equals: .startKm)

                Divider()

                // End KM
                InputRow(icon: "flag.checkered", iconColor: .red,
                         label: "Краен километраж", placeholder: "напр. 45450", unit: "км",
                         text: $vm.endKmText)
                    .focused($focused, equals: .endKm)

                // Distance badge / error
                Group {
                    if let d = vm.distance {
                        HStack(spacing: 6) {
                            Image(systemName: "arrow.left.and.right.circle.fill")
                            Text("Изминати: \(String(format: "%.1f", d)) км")
                                .fontWeight(.medium)
                        }
                        .font(.caption)
                        .foregroundStyle(.blue)
                        .padding(.horizontal, 10).padding(.vertical, 6)
                        .background(Color.blue.opacity(0.1))
                        .clipShape(RoundedRectangle(cornerRadius: 8))
                        .transition(.scale(scale: 0.85).combined(with: .opacity))
                    } else if vm.distanceError {
                        HStack(spacing: 6) {
                            Image(systemName: "exclamationmark.triangle.fill")
                            Text("Крайният км трябва да е по-голям от началния")
                        }
                        .font(.caption).foregroundStyle(.orange)
                        .padding(.horizontal, 10).padding(.vertical, 6)
                        .background(Color.orange.opacity(0.1))
                        .clipShape(RoundedRectangle(cornerRadius: 8))
                        .transition(.scale(scale: 0.85).combined(with: .opacity))
                    }
                }
                .animation(.spring(response: 0.3, dampingFraction: 0.8),
                           value: vm.distance != nil || vm.distanceError)

                Divider()

                InputRow(icon: "drop.fill", iconColor: .cyan,
                         label: "Заредени литри", placeholder: "напр. 40.5", unit: "л",
                         text: $vm.litersText)
                    .focused($focused, equals: .liters)

                Divider()

                InputRow(icon: "banknote.fill", iconColor: .green,
                         label: "Цена за литър", placeholder: "напр. 2.89", unit: "лв",
                         text: $vm.priceText)
                    .focused($focused, equals: .price)
            }
        }
        .onTapGesture { focused = nil }
    }
}

// MARK: - History Section

struct HistorySection: View {
    @ObservedObject var vm: FuelViewModel

    var body: some View {
        VStack(alignment: .leading, spacing: 12) {
            HStack {
                Label("История", systemImage: "clock.fill").font(.headline)
                Spacer()
                if !vm.history.isEmpty {
                    Text("\(vm.history.count) записа")
                        .font(.caption).foregroundStyle(.secondary)
                }
            }
            .padding(.horizontal, 4)

            if vm.history.isEmpty {
                EmptyHistoryView()
            } else {
                VStack(spacing: 10) {
                    ForEach(vm.history) { record in
                        HistoryRow(record: record, onDelete: { vm.delete(record) })
                            .transition(.asymmetric(
                                insertion: .move(edge: .top).combined(with: .opacity),
                                removal:   .move(edge: .trailing).combined(with: .opacity)
                            ))
                    }
                }
            }
        }
    }
}

// MARK: - History Row

struct HistoryRow: View {
    let record: FuelRecord
    let onDelete: () -> Void

    var consumptionColor: Color {
        switch record.consumption {
        case ..<6:  return .green
        case 6..<9: return .orange
        default:    return .red
        }
    }

    var body: some View {
        CardContainer {
            VStack(spacing: 10) {
                // Top row
                HStack {
                    Image(systemName: "calendar").font(.caption).foregroundStyle(.secondary)
                    Text(record.dateFormatted).font(.caption).foregroundStyle(.secondary)
                    Spacer()
                    ConsumptionBadge(value: record.consumption, color: consumptionColor)
                    Button(action: onDelete) {
                        Image(systemName: "trash")
                            .font(.caption).foregroundStyle(.red)
                    }
                    .buttonStyle(.plain)
                }

                Divider()

                // Odometer range
                HStack(spacing: 4) {
                    Image(systemName: "flag.fill").font(.caption2).foregroundStyle(.green)
                    Text(String(format: "%.0f", record.startKm))
                        .font(.caption.weight(.medium))
                    Image(systemName: "arrow.right").font(.caption2).foregroundStyle(.secondary)
                    Image(systemName: "flag.checkered").font(.caption2).foregroundStyle(.red)
                    Text(String(format: "%.0f", record.endKm))
                        .font(.caption.weight(.medium))
                    Text("км")
                        .font(.caption2).foregroundStyle(.secondary)
                    Spacer()
                    Text(String(format: "%.1f км", record.distance))
                        .font(.caption.weight(.semibold))
                        .foregroundStyle(.blue)
                }

                // Stats
                HStack(spacing: 0) {
                    StatItem(icon: "drop.fill",      color: .cyan,   value: String(format: "%.1f", record.liters),        unit: "л")
                    Spacer()
                    StatItem(icon: "banknote.fill",  color: .green,  value: String(format: "%.2f", record.pricePerLiter), unit: "лв/л")
                    Spacer()
                    StatItem(icon: "creditcard.fill",color: .orange, value: String(format: "%.2f", record.totalCost),     unit: "лв")
                }
            }
        }
    }
}
